export function optimizeResponse(response) {
  if (!response || !response.reply) return response;
  let reply = response.reply;
  // --- 1. Trim excessive length ---
  if (reply.length > 800) {
    reply = reply.slice(0, 800) + "...";
  }
  // --- 2. Split into readable chunks ---
  reply = reply
    .replace(/\n{3,}/g, "\n\n")   // normalize spacing
    .trim();
  // --- 3. Ensure short opening (first sentence only) ---
  const sentences = reply.split(/(?<=[.!?])\s+/);
  if (sentences.length > 3) {
    reply = [
      sentences[0], // validation
      sentences[1], // direction
      sentences.slice(2).join(" ")
    ].join(" ");
  }
  // --- 4. Add gentle structure if tool exists ---
  if (response.tool?.exercise?.steps) {
    const steps = response.tool.exercise.steps
      .map((s, i) => `${i + 1}. ${s}`)
      .join("\n");
    reply += `\n\nTry this:\n${steps}`;
  }
  return {
    ...response,
    reply
  };
}
