🔧 FIX — Add inference layer (no AI rewrite)

Create:

cat > server/ai/emotionInference.mjs <<'EOF'
export function inferEmotionalState(text = "") {
  const t = text.toLowerCase();
  const signals = {
    anxiety: /overthink|what if|racing|can't stop thinking/.test(t),
    overwhelm: /too much|everything|can't handle/.test(t),
    sadness: /tired of|pointless|nothing matters/.test(t),
    avoidance: /i'll deal later|ignore|don't want to/.test(t),
    self_doubt: /i always mess up|not good enough/.test(t),
  };
  return Object.keys(signals).filter(k => signals[k]);
}
EOF
