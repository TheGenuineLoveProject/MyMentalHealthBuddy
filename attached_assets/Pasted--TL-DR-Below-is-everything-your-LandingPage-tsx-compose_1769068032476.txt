🌀 2. SacredPattern.tsx — Animated Sacred SVG Background (Pulsing Geometry)

// SacredPattern.tsx
import React from 'react';
import styled, { keyframes } from 'styled-components';

const pulse = keyframes`
  0%, 100% {
    transform: scale(1);
    opacity: 0.15;
  }
  50% {
    transform: scale(1.05);
    opacity: 0.25;
  }
`;

const PatternWrapper = styled.div`
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  pointer-events: none;
  overflow: hidden;
  z-index: -1;
`;

const AnimatedSVG = styled.svg`
  position: absolute;
  width: 80vmin;
  height: 80vmin;
  top: 50%;
  left: 50%;
  transform: translate(-50%, -50%);
  animation: ${pulse} 6s ease-in-out infinite;
  fill: none;
  stroke: rgba(255, 255, 255, 0.08);
  stroke-width: 1;
`;

const SacredPattern = () => (
  <PatternWrapper>
    <AnimatedSVG viewBox="0 0 100 100" preserveAspectRatio="xMidYMid meet">
      <circle cx="50" cy="50" r="30" />
      <circle cx="50" cy="50" r="20" />
      <circle cx="50" cy="50" r="10" />
      <polygon points="50,15 85,85 15,85" />
    </AnimatedSVG>
  </PatternWrapper>
);

export default SacredPattern;
