---BEGIN PASTE (client/src/content/hubs/hubGraph.ts)---
import { HUBS } from "./topicHubs";
import { listAllRouteMeta, resolveRoutePath } from "../meta/routeMetaRegistry";

export type HubLink = { label: string; routeKey: string; href: string; reason?: string };

function overlap(a: string[] = [], b: string[] = []) {
  const A = new Set(a.map((x) => x.toLowerCase()));
  return b.reduce((n, x) => n + (A.has(x.toLowerCase()) ? 1 : 0), 0);
}

export function getHubMeta(topic) {
  return HUBS.find((h) => h.topic === topic);
}

export function getHubLinks(topic, limit = 18): HubLink[] {
  const hub = getHubMeta(topic);
  if (!hub) return [];

  const all = listAllRouteMeta();

  const scored = all
    .filter((m) => !m.isHub) // hubs list tools/paths, not other hubs
    .map((m) => ({
      m,
      score: overlap(hub.tags || [], m.tags || []),
    }))
    .filter((x) => x.score > 0)
    .sort((a, b) => (b.m.priority || 0) - (a.m.priority || 0) || b.score - a.score)
    .slice(0, limit)
    .map((x) => ({
      label: x.m.title,
      routeKey: x.m.routeKey,
      href: resolveRoutePath(x.m.routeKey),
      reason: "Matches hub topic",
    }));

  return scored;
}

const KEY = "glp:saves:v1";

function read(): string[] {
  try {
    return JSON.parse(localStorage.getItem(KEY) || "[]");
  } catch {
    return [];
  }
}

function write(v: string[]) {
  try {
    localStorage.setItem(KEY, JSON.stringify(v));
  } catch {}
}

export function isSaved(routeKey: string): boolean {
  return read().includes(routeKey);
}

export function toggleSave(routeKey: string) {
  const v = read();
  if (v.includes(routeKey)) write(v.filter((x) => x !== routeKey));
  else write([routeKey, ...v].slice(0, 500));
}

export function listSaves(): string[] {
  return read();
}

import { isSaved, toggleSave } from "../../content/saves/savesStore";

export function SaveButton({ routeKey }) {
  const [saved, setSaved] = useState(false);

  useEffect(() => {
    try {
      setSaved(isSaved(routeKey));
    } catch {
      setSaved(false);
    }
  }, [routeKey]);

  return (
    <button
      onClick={() => {
        toggleSave(routeKey);
        setSaved((s) => !s);

      className="rounded-xl border border-white/10 bg-white/10 px-3 py-2 text-sm hover:bg-white/15 transition"
    >
      {saved ? "Saved ★" : "Save"}

  );
}

- Produce JSON suggestions (DO NOT auto-edit registry by default)
- Output: scripts/.registry-suggestions.json

PASTE:

---BEGIN PASTE (scripts/generate-registry.mjs)---
import fs from "node:fs";
import path from "node:path";

const ROOT = process.cwd();
const PAGES_DIR = path.join(ROOT, "client", "src", "pages");
const OUT = path.join(ROOT, "scripts", ".registry-suggestions.json");

function walk(dir, out = []) {
  if (!fs.existsSync(dir)) return out;
  for (const ent of fs.readdirSync(dir, { withFileTypes: true })) {
    const p = path.join(dir, ent.name);
    if (ent.isDirectory()) walk(p, out);
    else if (/\.(tsx|ts|jsx|js)$/.test(ent.name)) out.push(p);
  }
  return out;
}

function slugify(s) {
  return s
    .replace(/\.(tsx|ts|jsx|js)$/i, "")
    .replace(/index$/i, "")
    .replace(/[^a-zA-Z0-9]+/g, "_")
    .replace(/^_+|_+$/g, "")
    .toLowerCase();
}

function deriveRouteKeyFromFile(filePath) {
  const rel = path.relative(PAGES_DIR, filePath).replace(/\\/g, "/");
  // Example: hubs/TopicHubPage.jsx -> hubs__topichubpage
  const parts = rel.split("/");
  const base = slugify(parts.join("__"));
  return base || "page__unknown";
}

function suggestCanonicalPath(filePath) {
  const rel = path.relative(PAGES_DIR, filePath).replace(/\\/g, "/");
  const parts = rel.split("/");
  // If file under hubs/, canonical uses /hubs/:topic for TopicHubPage; leave placeholder
  if (parts[0] === "hubs") return "/hubs/:topic";
  if (parts[0] === "dashboard") return "/dashboard/progress";
  if (parts[0] === "library") return "/library/saved";
  // default guess
  const p = "/" + rel.replace(/\.(tsx|ts|jsx|js)$/i, "").replace(/index$/i, "");
  return p.replace(/\/+$/g, "") || "/";
}

const files = walk(PAGES_DIR);
const suggestions = files.map((f) => ({
  file: path.relative(ROOT, f).replace(/\\/g, "/"),
  routeKey: deriveRouteKeyFromFile(f),
  canonicalPath: suggestCanonicalPath(f),
  title: null,
  description: null,
  benefits: [],
  internalLinks: [],
  modules: [],
  tags: [],
}));

fs.mkdirSync(path.dirname(OUT), { recursive: true });
fs.writeFileSync(OUT, JSON.stringify({ generatedAt: new Date().toISOString(), count: suggestions.length, suggestions }, null, 2));
console.log(`Wrote ${suggestions.length} suggestions -> ${path.relative(ROOT, OUT)}`);
---END PASTE---

Add npm script (package.json) if you have scripts section:
