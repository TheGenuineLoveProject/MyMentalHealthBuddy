Safe pattern

router.post("/chat", async (req, res) => {
  try {
    const { message } = req.body || {};
    if (!message || !String(message).trim()) {
      return res.status(400).json({ message: "Message is required" });
    }
    const text = String(message).trim();
    const userId = req.session?.user?.id || null;
    // Save user message only if authenticated
    if (userId) {
      await addMessage(userId, "user", text);
    }
    // existing AI / fallback logic here
    const reply = await generateReply(text);
    // Save assistant message only if authenticated
    if (userId) {
      await addMessage(userId, "assistant", reply);
    }
    return res.json({
      reply,
      authenticated: Boolean(userId),
    });
  } catch (error) {
    console.error("AI chat error:", error);
    return res.status(500).json({ message: "AI chat failed" });
  }
});

Exact Frontend Patch Pattern

In your chat UI file, make sure it calls the right route and handles non-200 errors.

Use this pattern:

async function sendMessage(message) {
  const res = await fetch("/api/ai/chat", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
    },
    credentials: "include",
    body: JSON.stringify({ message }),
  });
  const data = await res.json();
  if (!res.ok) {
    throw new Error(data.message || "Chat request failed");
  }
  return data.reply;
}

Phase 2 — Crisis Detection

Once Phase 1 works in browser, add crisis detection formally.

Exact backend logic target

In server/routes/ai.js or .mjs, add a crisis detector before normal reply generation.

Pattern

function detectCrisis(text = "") {
  const normalized = text.toLowerCase();
  const highRiskPatterns = [
    "i want to kill myself",
    "i want to die",
    "suicide",
    "end my life",
    "hurt myself",
    "self harm",
  ];
  return highRiskPatterns.some((pattern) => normalized.includes(pattern));
}

Then in route:

if (detectCrisis(text)) {
  const reply = "I'm really sorry you're going through this. You deserve immediate support right now. If you're in immediate danger, call emergency services now. If you're in the U.S. or Canada, call or text 988 right now for immediate crisis support. If you're elsewhere, contact your local emergency number or nearest crisis line now. If you can, please tell a trusted person near you immediately and do not stay alone.";
  if (userId) {
    await addMessage(userId, "assistant", reply);
  }
  return res.json({
    reply,
    safety: { flagged: true, type: "crisis" },
    action: "escalate_immediately",
  });
}

Response structure

function buildSupportiveReply(text) {
  const normalized = text.toLowerCase();
  if (normalized.includes("overwhelmed") || normalized.includes("anxious")) {
    return "It's completely understandable to feel anxious and overwhelmed, especially when there's a lot on your plate. It sounds like your mind may be carrying too many things at once right now. One gentle next step could be to pause and take three slow breaths, then choose just one small task that feels most manageable. What feels heaviest right now?";
  }
  return "Hi there—thanks for reaching out. How are you feeling today? What's on your mind? I'm here to listen without judgment. If you'd like, we can also try a quick grounding or coping exercise to help you feel a little more steady.";
}

This gives you a simple structured therapy-support layer without overbuilding.
