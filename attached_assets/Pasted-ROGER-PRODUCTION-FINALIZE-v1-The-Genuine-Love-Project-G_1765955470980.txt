function pascalCase(filePath) {
  const base = path.basename(filePath).replace(/\.(tsx|ts|jsx|js)$/, "");
  return base.replace(/(^|[-_ ]+)([a-zA-Z0-9])/g, (_, __, c) => c.toUpperCase());
}

const root = process.cwd();

let created = 0;
for (const rel of files) {
  const abs = path.join(root, rel);
  if (!fs.existsSync(abs)) {
    fs.mkdirSync(path.dirname(abs), { recursive: true });
    fs.writeFileSync(abs, template(pascalCase(rel)), "utf8");
    created++;
  }
}

console.log(`[figma:map] ensured ${files.length} files, created ${created}.`);
