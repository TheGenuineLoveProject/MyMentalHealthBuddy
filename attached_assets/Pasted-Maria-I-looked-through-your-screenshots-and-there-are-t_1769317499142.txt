I) Add an “Active Scan” script (detect incomplete pages + missing disclaimers)

Create: scripts/scanPlatform.mjs

#!/usr/bin/env node
/**
 * Scan platform pages for incompleteness + safety/SEO coverage.
 * Human-triggered only. Read-only scan; no file writes.
 */
import fs from "node:fs";
import path from "node:path";

const ROOT = process.cwd();
const PAGES_DIR = path.join(ROOT, "client", "src", "pages");

const REQUIRED_SNIPPETS = [
  { name: "SEO", pattern: /<SEO\b/ },
  { name: "SafetyFooter", pattern: /<SafetyFooter\b/ },
];

const DISALLOWED_PATTERNS = [
  { name: "TODO", pattern: /\bTODO\b/i },
  { name: "lorem", pattern: /\blorem ipsum\b/i },
  { name: "placeholder", pattern: /\bplaceholder\b/i },
  { name: "return null", pattern: /return\s+null\s*;/ },
];

const AGE_18_PATTERN =
  /(18\+|ages?\s*18|adults?\s*\(18\+\)|for\s+adults)/i;

function walk(dir) {
  const out = [];
  for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
    const p = path.join(dir, entry.name);
    if (entry.isDirectory()) out.push(...walk(p));
    else if (/\.(tsx|jsx|ts|js)$/.test(entry.name)) out.push(p);
  }
  return out;
}

function read(p) {
  return fs.readFileSync(p, "utf8");
}

const files = fs.existsSync(PAGES_DIR) ? walk(PAGES_DIR) : [];
if (!files.length) {
  console.error(`No pages found at ${PAGES_DIR}`);
  process.exit(1);
}

const report = [];
for (const file of files) {
  const src = read(file);
  const rel = path.relative(ROOT, file);

  const missing = REQUIRED_SNIPPETS.filter((r) => !r.pattern.test(src)).map(
    (r) => r.name
  );

  const bad = DISALLOWED_PATTERNS.filter((d) => d.pattern.test(src)).map(
    (d) => d.name
  );

  const has18 = AGE_18_PATTERN.test(src);

  const flags = [];
  if (missing.length) flags.push(`Missing: ${missing.join(", ")}`);
  if (!has18) flags.push(`Missing: 18+ language (or ensure global disclaimer is rendered)`);
  if (bad.length) flags.push(`Contains: ${bad.join(", ")}`);

  if (flags.length) report.push({ file: rel, flags });
}

if (!report.length) {
  console.log("✅ Scan complete: no issues detected in pages.");
  process.exit(0);
}

console.log("⚠️ Scan complete: issues found.\n");
for (const item of report) {
  console.log(`- ${item.file}`);
  for (const f of item.flags) console.log(`   • ${f}`);
}
console.log(`\nTotal flagged: ${report.length}/${files.length}`);
process.exit(2);

Add to package.json:

{
  "scripts": {
    "scan:platform": "node scripts/scanPlatform.mjs"
  }
}

Run:

npm run scan:platform
