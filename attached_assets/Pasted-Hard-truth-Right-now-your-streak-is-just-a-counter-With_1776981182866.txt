STEP 1 — Add streak decay (prevents false streaks)

Edit: server/routes/streaks.mjs

Find your check-in logic and replace ONLY the streak calculation block with:

const today = new Date();
const last = user.lastCheckin ? new Date(user.lastCheckin) : null;
let streak = user.streak || 0;
if (!last) {
  streak = 1;
} else {
  const diffDays = Math.floor((today - last) / (1000 * 60 * 60 * 24));
  if (diffDays === 0) {
    // already checked in today
  } else if (diffDays === 1) {
    streak += 1;
  } else {
    streak = 1; // reset if missed a day
  }
}
user.streak = streak;
user.lastCheckin = today;
