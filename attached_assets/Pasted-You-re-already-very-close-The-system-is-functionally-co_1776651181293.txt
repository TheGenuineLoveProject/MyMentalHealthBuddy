#!/usr/bin/env bash
set -e
echo "Checking contract route uniqueness..."
node scripts/generate-route-manifest.mjs > /dev/null
if [ ! -f docs/ROUTE_MANIFEST.json ]; then
  echo "❌ ROUTE_MANIFEST.json missing"
  exit 1
fi
node << 'EOF'
const fs = require('fs');
const routes = JSON.parse(fs.readFileSync('docs/ROUTE_MANIFEST.json', 'utf8'));
const seen = new Set();
for (const r of routes) {
  const key = `${r.method}:${r.path}`;
  if (seen.has(key)) {
    console.error("❌ Duplicate route:", key);
    process.exit(1);
  }
  seen.add(key);
}
console.log("✅ Route contract valid");
EOF
