import { useMemo } from "react";
import { getRouteMeta } from "../../content/meta/routeMetaRegistry";

function keyFor(routeKey) {
  return `glp:saved:${routeKey}`;
}

export function SavedToolsButton({ routeKey }) {
  const meta = useMemo(() => getRouteMeta(routeKey), [routeKey]);

  function toggle() {
    try {
      const k = keyFor(routeKey);
      const existing = localStorage.getItem(k);
      if (existing) {
        localStorage.removeItem(k);
        alert("Removed from saved");
      } else {
        localStorage.setItem(k, JSON.stringify({ routeKey, title: meta.title, canonicalPath: meta.canonicalPath, savedAt: Date.now() }));
        alert("Saved ✅");
      }
    } catch {
      alert("Could not save. Your browser may be blocking storage.");
    }
  }

  return (
    <button
      onClick={toggle}
      className="rounded-xl border border-white/10 bg-white/10 px-3 py-2 text-sm hover:bg-white/15 transition"
    >
      Save this page
    </button>
  );
}
---END PASTE---
