HeroSection.tsx — ✅ Healing Hero in React + Styled Components (with subtle effects)

// HeroSection.tsx
import React from 'react';
import styled from 'styled-components';
import { brandColors, brandFonts, gradients, shadows } from './theme';

const HeroContainer = styled.section`
  background: ${gradients.radiantHealing};
  min-height: 100vh;
  display: flex;
  flex-direction: column;
  justify-content: center;
  align-items: center;
  padding: 4rem 2rem;
  color: ${brandColors.charcoal};
  font-family: ${brandFonts.primary};
  text-align: center;
  animation: fadeIn 2s ease-in;

  @keyframes fadeIn {
    from { opacity: 0; transform: translateY(10px); }
    to { opacity: 1; transform: translateY(0); }
  }
`;

const Logo = styled.div`
  width: 120px;
  height: 120px;
  background: url('/logo.svg') no-repeat center;
  background-size: contain;
  margin-bottom: 2rem;
`;

const Title = styled.h1`
  font-family: ${brandFonts.display};
  font-size: 3rem;
  margin-bottom: 1rem;
  color: ${brandColors.charcoal};
`;

const Subtitle = styled.p`
  max-width: 600px;
  font-size: 1.25rem;
  color: ${brandColors.deepTeal};
`;

const CallToAction = styled.a`
  margin-top: 2rem;
  padding: 0.75rem 2rem;
  background: ${brandColors.deepTeal};
  color: ${brandColors.softWhite};
  border-radius: 8px;
  font-weight: 600;
  box-shadow: ${shadows.softGlow};
  text-decoration: none;
  transition: all 0.3s ease;
  
  &:hover {
    background: ${brandColors.sageGreen};
    transform: scale(1.05);
  }
`;

const HeroSection = () => (
  <HeroContainer>
    <Logo />
    <Title>The Genuine Love Project</Title>
    <Subtitle>
      Healing hearts. Inspiring minds. Awakening love—through sacred design and conscious visuals.
    </Subtitle>
    <CallToAction href="#explore">Begin Your Journey</CallToAction>
  </HeroContainer>
);

export default HeroSection;
