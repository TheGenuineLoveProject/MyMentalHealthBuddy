--- FILE: client/src/components/wellness/BenefitsBlock.tsx ---
import React from "react";

export type BenefitsTier = "beginner" | "intermediate" | "advanced";

export interface BenefitItem {
  title: string;
  body: string;
}

export interface BenefitsBlockProps {
  title?: string;
  tier?: BenefitsTier; // default: "beginner"
  items: BenefitItem[];
}

const tierClassMap: Record<BenefitsTier, string> = {
  beginner: "glp-benefits--beginner",
  intermediate: "glp-benefits--intermediate",
  advanced: "glp-benefits--advanced",
};

export default function BenefitsBlock({
  title = "What you can do here",
  tier = "beginner",
  items,
}: BenefitsBlockProps) {
  return (
    <section className={`glp-benefits ${tierClassMap[tier]}`} aria-labelledby="benefits-title">
      <h2 id="benefits-title" className="glp-benefits__title">
        {title}
      </h2>

      <ul className="glp-benefits__list">
        {items.map((item, i) => (
          <li key={i} className="glp-benefits__item">
            <strong className="glp-benefits__item-title">{item.title}</strong>
            <p className="glp-benefits__item-body">{item.body}</p>
          </li>
        ))}
      </ul>
    </section>
  );
}
