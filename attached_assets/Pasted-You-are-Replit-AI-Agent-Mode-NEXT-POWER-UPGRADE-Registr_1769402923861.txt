import { useEffect, useMemo, useState } from "react";

function clamp(n, min, max) {
  const x = Number.isFinite(Number(n)) ? Number(n) : min;
  return Math.max(min, Math.min(max, x));
}

function keyFor(routeKey) {
  return `glp:mi:${routeKey}`;
}

export function MIBlock({ routeKey }) {
  const storageKey = useMemo(() => keyFor(routeKey), [routeKey]);
  const [importance, setImportance] = useState(6);
  const [confidence, setConfidence] = useState(5);
  const [whyNow, setWhyNow] = useState("");
  const [tinyAction, setTinyAction] = useState("");

  useEffect(() => {
    try {
      const raw = localStorage.getItem(storageKey);
      if (!raw) return;
      const saved = JSON.parse(raw);
      if (saved?.importance != null) setImportance(clamp(saved.importance, 0, 10));
      if (saved?.confidence != null) setConfidence(clamp(saved.confidence, 0, 10));
      if (saved?.whyNow != null) setWhyNow(String(saved.whyNow));
      if (saved?.tinyAction != null) setTinyAction(String(saved.tinyAction));
    } catch {}
  }, [storageKey]);

  useEffect(() => {
    try {
      localStorage.setItem(
        storageKey,
        JSON.stringify({ importance, confidence, whyNow, tinyAction })
      );
    } catch {}
  }, [storageKey, importance, confidence, whyNow, tinyAction]);

  const oars = [
    "Open question: What matters most to you about this?",
    "Affirmation: What strength have you already shown?",
    "Reflection: If your best friend said this, what would you hear beneath the words?",
    "Summary: In one sentence, what are you choosing next?",
  ];

  return (
    <section className="rounded-2xl border border-white/10 bg-black/10 p-4">
      <div className="flex items-center justify-between gap-3">
        <h3 className="text-sm font-semibold">Motivational Interviewing (MI)</h3>
        <span className="text-xs opacity-70">OARS + rulers</span>
      </div>

      <div className="mt-3 grid gap-3">
        <div className="text-xs opacity-80 leading-relaxed">
          Use these prompts to strengthen “change talk” (your reasons for taking a small step).
        </div>

        <div className="grid gap-2">
          <div className="text-xs font-semibold opacity-80">OARS prompts</div>
          <ul className="list-disc pl-5 text-sm opacity-90">
            {oars.map((p) => (
              <li key={p}>{p}</li>
            ))}
          </ul>
        </div>

        <div className="grid md:grid-cols-2 gap-3">
          <label className="grid gap-1">
            <div className="flex items-center justify-between">
              <span className="text-xs font-semibold opacity-80">Importance</span>
              <span className="text-xs opacity-70">{importance}/10</span>
            </div>
            <input
              type="range"
              min="0"
              max="10"
              value={importance}
              onChange={(e) => setImportance(clamp(e.target.value, 0, 10))}
            />
            <div className="text-xs opacity-70">
              Why is it a {importance} and not lower?
            </div>
          </label>

          <label className="grid gap-1">
            <div className="flex items-center justify-between">
              <span className="text-xs font-semibold opacity-80">Confidence</span>
              <span className="text-xs opacity-70">{confidence}/10</span>
            </div>
            <input
              type="range"
              min="0"
              max="10"
              value={confidence}
              onChange={(e) => setConfidence(clamp(e.target.value, 0, 10))}
            />
            <div className="text-xs opacity-70">
              What would move you one point higher?
            </div>
          </label>
        </div>

        <label className="grid gap-1">
          <span className="text-xs font-semibold opacity-80">Why now?</span>
          <textarea
            className="min-h-[70px] rounded-xl border border-white/10 bg-black/20 p-2 text-sm"
            value={whyNow}
            onChange={(e) => setWhyNow(e.target.value)}
            placeholder="Because… (one honest reason is enough)"
          />
        </label>

        <label className="grid gap-1">
          <span className="text-xs font-semibold opacity-80">One tiny action</span>
          <input
            className="rounded-xl border border-white/10 bg-black/20 p-2 text-sm"
            value={tinyAction}
            onChange={(e) => setTinyAction(e.target.value)}
            placeholder="Example: 2 minutes of breathing after I brush my teeth"
          />
          <div className="text-xs opacity-70">
            Keep it small enough to succeed on your hardest day.
          </div>
        </label>
      </div>
    </section>
  );
}

import { useEffect, useMemo, useState } from "react";

  return `glp:tinystep:${routeKey}`;
}

  const [step, setStep] = useState("");
  const [whenWhere, setWhenWhere] = useState("");

import { useMemo, useState } from "react";
import { getRouteMeta } from "../../../content/meta/routeMetaRegistry";

function keyFor(routeKey) {
  return `glp:share:${routeKey}`;
}

export function ShareCardBlock({ routeKey }) {
  const meta = useMemo(() => getRouteMeta(routeKey), [routeKey]);
  const [note, setNote] = useState("");

  const shareText = useMemo(() => {
    const title = meta?.title || "Genuine Love Practice";
    const line1 = `✨ 2-minute win: ${title}`;
    const line2 = `🫶 Today I’m choosing one small step (no pressure, just practice).`;
    const line3 = `🔁 Next: one tiny action + breathe + repeat.`;
    const line4 = `#GenuineLoveProject #GentleGrowth #TinySteps`;
    return [line1, line2, note ? `📝 ${note}` : null, line3, line4].filter(Boolean).join("\n");
  }, [meta, note]);

  async function copy() {
    try {
      await navigator.clipboard.writeText(shareText);
      try {
        localStorage.setItem(keyFor(routeKey), shareText);
      } catch {}
      alert("Copied ❤️");
    } catch {
      alert("Copy failed. You can manually select and copy.");
    }
  }

  return (
    <section className="rounded-2xl border border-white/10 bg-black/10 p-4">
      <div className="flex items-center justify-between gap-3">
        <h3 className="text-sm font-semibold">Infinity-Heart share</h3>
        <span className="text-xs opacity-70">optional</span>
      </div>

      <div className="mt-3 grid gap-3">
        <label className="grid gap-1">
          <span className="text-xs font-semibold opacity-80">Add a personal line (optional)</span>
          <input
            className="rounded-xl border border-white/10 bg-black/20 p-2 text-sm"
            value={note}
            onChange={(e) => setNote(e.target.value)}
            placeholder="Example: I’m proud I showed up today."
          />
        </label>

        <div className="rounded-xl border border-white/10 bg-black/20 p-3 text-sm whitespace-pre-line">
          {shareText}
        </div>

        <button
          onClick={copy}
          className="rounded-xl border border-white/10 bg-white/10 px-3 py-2 text-sm hover:bg-white/15 transition"
        >
          Copy share text
        </button>

        <div className="text-xs opacity-70">
          This is wellness content. No medical promises. Share only what feels safe.
        </div>
      </div>
    </section>
  );
}
