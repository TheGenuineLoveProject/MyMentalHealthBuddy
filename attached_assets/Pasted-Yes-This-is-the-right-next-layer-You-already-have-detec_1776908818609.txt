2) Create tool selector

File

server/ai/toolSelector.mjs

// server/ai/toolSelector.mjs
import { TOOLS } from "./tools.mjs";
export function selectTool({ modules = [], profile = {}, input = "" }) {
  if (!modules.length) return null;
  const names = modules.map(m => m.name);
  // simple priority order based on selected modules
  for (const modName of names) {
    const match = Object.values(TOOLS).find(t => t.module === modName);
    if (match) return match;
  }
  // profile-aware fallback
  if ((profile?.emotional_patterns || []).some(p => /anxiet/i.test(p))) {
    return TOOLS.box_breathing;
  }
  if (/overwhelm|too much|burnout/i.test(input)) {
    return TOOLS.overload_reset;
  }
  return null;
}

3) Create tool executor

File

server/ai/toolExecutor.mjs

// server/ai/toolExecutor.mjs
export function executeTool(tool, input = "") {
  if (!tool) return null;
  return {
    tool: {
      id: tool.id,
      title: tool.title,
      type: tool.type,
      durationMin: tool.durationMin
    },
    exercise: {
      intro: buildIntro(tool, input),
      steps: tool.steps,
      closing: buildClosing(tool)
    }
  };
}
function buildIntro(tool, input) {
  switch (tool.id) {
    case "box_breathing":
      return "Let’s slow things down together with one short breathing round.";
    case "grounding_54321":
      return "Let’s gently anchor you in the present moment.";
    case "thought_reframe":
      return "Let’s look at the thought carefully instead of letting it run the whole story.";
    case "emotional_checkin":
      return "Let’s slow down and notice what you’re feeling.";
    case "relationship_repair":
      return "Let’s prepare your thoughts so the conversation is clearer and kinder.";
    case "pattern_interrupt":
      return "Let’s identify the loop and create one interruption point.";
    case "overload_reset":
      return "Let’s reduce the load and choose one manageable next step.";
    default:
      return `Let's work through: ${tool.title}`;
  }
}
function buildClosing(tool) {
  return "Take what serves you. You know yourself best.";
}
