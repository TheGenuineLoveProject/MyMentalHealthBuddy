import { getRouteMeta, deriveRouteKeyFromPath } from "./meta/routeMetaRegistry";

/**
 * Single entry point for config used by Autopilot + PageTemplate.
 * No placeholders. Must be valid JS.
 */
export function getRouteConfig(pathname, opts = {}) {
  const routeKey = opts.routeKey || deriveRouteKeyFromPath(pathname);
  const meta = getRouteMeta(routeKey);

  return {
    routeKey,
    canonicalPath: meta.canonicalPath,
    title: meta.title,
    description: meta.description,
    benefits: meta.benefits,
    internalLinks: meta.internalLinks, // routeKeys only
    modules: meta.modules,
    protected: Boolean(meta.protected || opts.protected || false),
  };
}
