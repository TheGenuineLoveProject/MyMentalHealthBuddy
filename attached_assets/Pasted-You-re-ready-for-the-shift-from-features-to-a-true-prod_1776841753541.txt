// server/ai/moduleRouter.mjs
import { MODULES } from "./modules.mjs";
export function selectModules({ input = "", profile = {} }) {
  const selected = [];
  for (const [name, mod] of Object.entries(MODULES)) {
    if (mod.triggers.test(input)) {
      selected.push({ name, ...mod });
    }
  }
  // profile-based activation
  if (profile?.behavior_loops?.length) {
    selected.push({ name: "behavioral_loop", ...MODULES.behavioral_loop });
  }
  // sort by priority
  selected.sort((a, b) => b.priority - a.priority);
  return selected.slice(0, 2); // limit to top 2 modules
}
