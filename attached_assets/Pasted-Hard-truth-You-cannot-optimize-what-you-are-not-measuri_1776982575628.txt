2. Add behavior log file

Create:

cat > server/utils/behaviorLogger.mjs <<'EOF'
import fs from "fs";
import path from "path";
const DIR = path.resolve("data/logs");
const FILE = path.join(DIR, "behavior-events.jsonl");
function ensureDir() {
  if (!fs.existsSync(DIR)) fs.mkdirSync(DIR, { recursive: true });
}
export function logBehaviorEvent(event = {}) {
  try {
    ensureDir();
    const safe = {
      type: event.type || "unknown",
      guestId: event.guestId || null,
      toolId: event.toolId || null,
      module: event.module || null,
      streak: event.streak ?? null,
      daysAway: event.daysAway ?? null,
      paywallReason: event.paywallReason || null,
      timestamp: new Date().toISOString()
    };
    fs.appendFileSync(FILE, JSON.stringify(safe) + "\n");
  } catch {}
}
EOF

4. Analyze drop-off

After real users or your own tests:

tail -n 50 data/logs/behavior-events.jsonl

Look for:

start_page_view but no first_tool_selected = weak first screen
first_tool_selected but no first_response_success = slow/error response
first_response_success but no streak_incremented = weak habit loop
streak_incremented but no return_user_detected = weak reactivation
paywall_shown but no paywall_clicked = bad timing or weak offer
