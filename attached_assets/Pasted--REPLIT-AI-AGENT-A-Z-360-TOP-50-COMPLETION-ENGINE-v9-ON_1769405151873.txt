CONTENT:
  import fs from "node:fs";
  import path from "node:path";

  const ROOT = process.cwd();
  const ALLOWED = new Set(["Beginner", "Intermediate", "Advanced"]);

  // Forbidden phrases we never want again
  const FORBIDDEN_PATTERNS = [
    /reading\s*level/i,
    /grade\s*level/i,
    /reading_level/i,
    /readingLevel/i,
    /grade_level/i,
  ];

  const exts = new Set([".ts", ".tsx", ".js", ".jsx", ".md", ".json", ".mjs"]);
  const ignoreDirs = new Set(["node_modules", ".git", "dist", "build", ".next", ".cache"]);

  function walk(dir) {
    for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
      if (ignoreDirs.has(entry.name)) continue;
      const p = path.join(dir, entry.name);
      if (entry.isDirectory()) walk(p);
      else {
        const ext = path.extname(entry.name);
        if (!exts.has(ext)) continue;
        const text = fs.readFileSync(p, "utf8");

        for (const re of FORBIDDEN_PATTERNS) {
          if (re.test(text)) {
            console.error(`❌ Forbidden tier language found in: ${p}`);
            process.exitCode = 1;
            return;
          }
        }

        // Optional: If file explicitly mentions "tier" but none of allowed labels appear, warn.
        if (/tier/i.test(text) && !/(Beginner|Intermediate|Advanced)/.test(text)) {
          console.warn(`⚠️ Tier referenced without allowed labels in: ${p}`);
        }
      }
    }
  }

  walk(ROOT);

  if (process.exitCode === 1) {
    console.error("Fix: Replace any reading/grade language with Beginner/Intermediate/Advanced only.");
    process.exit(1);
  }

  console.log("✅ content-check passed (only allowed tiers & no reading/grade language).");
