// server/ai/profileStore.mjs
import fs from "fs";
import path from "path";
const PROFILE_DIR = path.resolve("data/user-profile");
function ensureDir() {
  if (!fs.existsSync(PROFILE_DIR)) {
    fs.mkdirSync(PROFILE_DIR, { recursive: true });
  }
}
function sanitize(userId) {
  return userId.replace(/[^a-zA-Z0-9_-]/g, "_");
}
function getFile(userId) {
  return path.join(PROFILE_DIR, `${sanitize(userId)}.json`);
}
export function loadProfile(userId) {
  try {
    ensureDir();
    const file = getFile(userId);
    if (!fs.existsSync(file)) {
      return {
        emotional_patterns: [],
        triggers: [],
        relationship_themes: [],
        coping_strategies: [],
        support_needs: [],
        risk_flags: []
      };
    }
    return JSON.parse(fs.readFileSync(file, "utf-8"));
  } catch {
    return {
      emotional_patterns: [],
      triggers: [],
      relationship_themes: [],
      coping_strategies: [],
      support_needs: [],
      risk_flags: []
    };
  }
}
export function saveProfile(userId, profile) {
  try {
    ensureDir();
    const file = getFile(userId);
    fs.writeFileSync(file, JSON.stringify(profile, null, 2));
  } catch (err) {
    console.warn("Profile save failed:", err.message);
  }
}

// server/ai/profileExtractor.mjs
export async function extractProfile({ openai, history }) {
  try {
    const text = history
      .map(m => `${m.role}: ${m.content}`)
      .join("\n");
    const response = await openai.chat.completions.create({
      model: "gpt-4o-mini",
      temperature: 0.2,
      messages: [
        {
          role: "system",
          content: `
Extract structured psychological insights.
Return ONLY valid JSON with:
- emotional_patterns (array)
- triggers (array)
- relationship_themes (array)
- coping_strategies (array)
- support_needs (array)
- risk_flags (array)
Be concise. No explanations.
          `.trim()
        },
        {
          role: "user",
          content: text
        }
      ]
    });
    const raw = response.choices?.[0]?.message?.content || "{}";
    return JSON.parse(raw);
  } catch (err) {
    console.warn("Profile extraction failed:", err.message);
    return null;
  }
}

🪜 STEP 4 — ADD MERGE FUNCTION

📍 ADD TO BOTTOM OF ORCHESTRATOR

function mergeProfiles(existing, incoming) {
  const result = { ...existing };
  for (const key of Object.keys(existing)) {
    const set = new Set([
      ...(existing[key] || []),
      ...(incoming[key] || [])
    ]);
    result[key] = Array.from(set).slice(-10);
  }
  return result;
}
