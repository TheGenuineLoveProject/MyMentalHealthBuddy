STEP 2 — Fetch streak count (lightweight)

Add in Start.tsx

const [streak, setStreak] = useState(0);
useEffect(() => {
  fetch("/api/streaks/me", {
    headers: { "x-guest-id": guestId }
  })
    .then(res => res.json())
    .then(data => setStreak(data.streak || 0))
    .catch(() => {});
}, []);

STEP 5 — Add “next-day hook” memory (light backend)

export function getReturnPrompt() {
  const prompts = [
    "How are you feeling today?",
    "Want a quick reset?",
    "Let’s take one minute for you.",
    "Check in with yourself today."
  ];
  return prompts[Math.floor(Math.random() * prompts.length)];
}
EOF
