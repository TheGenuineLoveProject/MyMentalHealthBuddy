export function selectModules({ profile, input }) {
  const modules = [];
  if (/anxious|panic|worry/i.test(input)) {
    modules.push("anxiety");
  }
  if (profile?.behavior_loops?.length) {
    modules.push("loop_detection");
  }
  if (/relationship|partner|fight/i.test(input)) {
    modules.push("relationship");
  }
  return modules;
}
