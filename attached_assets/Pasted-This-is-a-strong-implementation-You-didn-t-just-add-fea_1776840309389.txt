Next: Adaptive Response Layer (style + safety policy driven by profile & risk)

Goal: tailor tone, structure, and intervention style per user without leaking business logic into healing outputs.

⸻

🧠 What this layer does

It converts your signals into a response policy:

Inputs:

* risk.level (low/medium/high)
* profile (patterns, triggers, support_needs)
* summary (context)
* scoring.tier (standard/high)

Output:

* style contract (tone, verbosity, structure)
* intervention plan (grounding vs exploration vs psychoeducation)
* constraints (crisis brevity, avoid certain patterns, etc.)

Then it injects a single, controlled system message for the provider.

⸻

🪜 STEP 1 — Create policy builder

📁 server/ai/responsePolicy.mjs

// server/ai/responsePolicy.mjs
export function buildResponsePolicy({ risk, profile, scoring }) {
  const level = risk?.level || "low";
  const tier = scoring?.tier || "standard";
  // base defaults
  let tone = "warm, validating, non-clinical";
  let verbosity = tier === "high" ? "moderate" : "concise";
  let structure = "reflect → validate → 1–2 options";
  let constraints = [
    "no diagnosis",
    "no medical/legal advice",
    "avoid absolute claims",
  ];
  let interventions = ["reflection", "grounding", "small_next_step"];
  // risk escalation
  if (level === "high") {
    tone = "calm, steady, grounding";
    verbosity = "concise";
    structure = "acknowledge → immediate grounding → resources";
    interventions = ["grounding", "safety", "resource_signpost"];
    constraints.push("do not overwhelm with options");
  }
  // profile-informed tweaks
  const needs = new Set(profile?.support_needs || []);
  if (needs.has("reassurance")) {
    interventions.push("reassurance");
  }
  if ((profile?.emotional_patterns || []).some(p => /anxiet/i.test(p))) {
    interventions.push("breathing", "5-4-3-2-1");
  }
  if ((profile?.risk_flags || []).length > 0) {
    constraints.push("monitor for escalation cues");
  }
  return {
    tone,
    verbosity,
    structure,
    interventions,
    constraints,
  };
}
export function renderPolicySystemMessage(policy) {
  const {
    tone, verbosity, structure, interventions, constraints
  } = policy;
  return [
    "Response policy:",
    `- Tone: ${tone}`,
    `- Verbosity: ${verbosity}`,
    `- Structure: ${structure}`,
    `- Interventions: ${interventions.join(", ")}`,
    `- Constraints: ${constraints.join("; ")}`
  ].join("\n");
}

  // summary (if any)

  // NEW: response policy (last system message before conversation)

    role: "system",
    content: policyMsg
