server/ai/paywallPolicy.mjs

// server/ai/paywallPolicy.mjs
export function shouldShowPaywall({
  isCrisis = false,
  successfulSessions = 0,
  dailySessions = 0,
  requestedPremium = false
}) {
  if (isCrisis) return false;
  if (requestedPremium) return true;
  if (successfulSessions >= 3) return true;
  if (dailySessions >= 3) return true;
  return false;
}
export function getPaywallReason({
  successfulSessions = 0,
  dailySessions = 0,
  requestedPremium = false
}) {
  if (requestedPremium) return "premium_feature";
  if (successfulSessions >= 3) return "value_proven";
  if (dailySessions >= 3) return "daily_limit";
  return null;
}

server/utils/usageCounter.mjs

Start simple and file-backed if you want smallest-safe implementation.

// server/utils/usageCounter.mjs
import fs from "fs";
import path from "path";
const DIR = path.resolve("data/usage");
function ensureDir() {
  if (!fs.existsSync(DIR)) fs.mkdirSync(DIR, { recursive: true });
}
function sanitize(id) {
  return String(id || "anonymous").replace(/[^a-zA-Z0-9_-]/g, "_");
}
function getFile(id) {
  return path.join(DIR, `${sanitize(id)}.json`);
}
export function readUsage(id) {
  try {
    ensureDir();
    const file = getFile(id);
    if (!fs.existsSync(file)) {
      return {
        successfulSessions: 0,
        dailySessions: 0,
        lastDay: new Date().toISOString().slice(0, 10)
      };
    }
    const data = JSON.parse(fs.readFileSync(file, "utf-8"));
    const today = new Date().toISOString().slice(0, 10);
    if (data.lastDay !== today) {
      data.dailySessions = 0;
      data.lastDay = today;
    }
    return data;
  } catch {
    return {
      successfulSessions: 0,
      dailySessions: 0,
      lastDay: new Date().toISOString().slice(0, 10)
    };
  }
}
export function incrementSuccessfulSession(id) {
  const data = readUsage(id);
  data.successfulSessions += 1;
  data.dailySessions += 1;
  ensureDir();
  fs.writeFileSync(getFile(id), JSON.stringify(data, null, 2));
  return data;
}

Safe backend wiring

In server/routes/ai.mjs

After a successful non-crisis response, increment usage and attach paywall hint.

Add imports:

import { incrementSuccessfulSession } from "../utils/usageCounter.mjs";
import { shouldShowPaywall, getPaywallReason } from "../ai/paywallPolicy.mjs";

Then before final res.json(...) on successful non-crisis response:

const identity =
  req.dbUserId ||
  req.headers["x-guest-id"] ||
  "anonymous";
const isCrisis = Boolean(result?.response?.isCrisis);
if (result?.ok && !isCrisis && result?.response?.reply) {
  const usage = incrementSuccessfulSession(identity);
  result.response.paywall = {
    show: shouldShowPaywall({
      isCrisis,
      successfulSessions: usage.successfulSessions,
      dailySessions: usage.dailySessions
    }),
    reason: getPaywallReason({
      successfulSessions: usage.successfulSessions,
      dailySessions: usage.dailySessions
    })
  };
}

This keeps the decision on the server, which is safer.
