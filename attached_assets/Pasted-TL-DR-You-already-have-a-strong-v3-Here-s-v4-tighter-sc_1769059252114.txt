--- tools/release_gate.sh ---
#!/usr/bin/env bash
set -euo pipefail

ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
cd "$ROOT"

echo "== Release Gate: TheGenuineLoveProject =="
date
echo "Repo: $ROOT"
echo

have_cmd() { command -v "$1" >/dev/null 2>&1; }
have_file() { [ -f "$1" ]; }

run() {
  local label="$1"; shift
  echo "== $label =="
  "$@"
  echo
}

node_pm() {
  if have_file "pnpm-lock.yaml" && have_cmd pnpm; then echo "pnpm"; return; fi
  if have_file "yarn.lock" && have_cmd yarn; then echo "yarn"; return; fi
  if have_file "package-lock.json" && have_cmd npm; then echo "npm_ci"; return; fi
  if have_file "package.json" && have_cmd npm; then echo "npm"; return; fi
  echo ""
}

has_npm_script() {
  local script="$1"
  node -e "const p=require('./package.json'); process.exit(p.scripts && p.scripts['$script'] ? 0 : 1)" >/dev/null 2>&1
}

run_npm_script_if() {
  local script="$1"
  if has_npm_script "$script"; then run "npm run $script" npm run -s "$script"; else echo "SKIP: $script"; echo; fi
}

PM="$(node_pm)"

if [ -n "$PM" ]; then
  echo "== Node detected ($PM) =="
  if [ "$PM" = "pnpm" ]; then run "pnpm install" pnpm install;
  elif [ "$PM" = "yarn" ]; then run "yarn install" yarn install --frozen-lockfile || yarn install;
  elif [ "$PM" = "npm_ci" ]; then run "npm ci" npm ci;
  else run "npm install" npm install; fi

  run_npm_script_if lint
  run_npm_script_if typecheck
  run_npm_script_if test
  run_npm_script_if build

  if have_file "tools/env_audit.js"; then run "env audit" node tools/env_audit.js; fi
  if have_file "tools/route_map.js"; then run "route map" node tools/route_map.js; fi
  if have_file "tools/link_scan.js"; then run "link scan" node tools/link_scan.js; fi
  if have_file "tools/auth_smoke.js"; then run "auth smoke" node tools/auth_smoke.js; fi
else
  echo "== No Node package.json detected. =="
  echo
fi

if have_file "requirements.txt" && have_cmd python; then
  run "pip install" python -m pip install -r requirements.txt
  if have_cmd pytest; then run "pytest" pytest -q; fi
fi

echo "== RELEASE GATE: PASS =="
exit 0


--- tools/env_audit.js ---
/**
 * Env audit: discovers env usage patterns and checks .env.example coverage.
 */
const fs = require("fs");
const path = require("path");

const ROOT = path.resolve(__dirname, "..");
const EXCLUDE = new Set(["node_modules", ".git", "dist", "build", ".next", ".cache"]);

const PATTERNS = [
  /process\.env\.([A-Z0-9_]+)/g,
  /import\.meta\.env\.([A-Z0-9_]+)/g,
];

function walk(dir, out = []) {
  for (const ent of fs.readdirSync(dir, { withFileTypes: true })) {
    if (ent.isDirectory()) {
      if (EXCLUDE.has(ent.name)) continue;
      walk(path.join(dir, ent.name), out);
    } else if (ent.isFile()) {
      const p = path.join(dir, ent.name);
      if (/\.(js|jsx|ts|tsx|mjs|cjs|md|mdx|py)$/.test(p)) out.push(p);
    }
  }
  return out;
}

function parseDotEnv(fp) {
  if (!fs.existsSync(fp)) return new Set();
  const keys = new Set();
  for (const line of fs.readFileSync(fp, "utf8").split(/\r?\n/)) {
    const t = line.trim();
    if (!t || t.startsWith("#")) continue;
    const i = t.indexOf("=");
    if (i > 0) keys.add(t.slice(0, i).trim());
  }
  return keys;
}

const used = new Set();
for (const f of walk(ROOT)) {
  const txt = fs.readFileSync(f, "utf8");
  for (const re of PATTERNS) {
    let m;
    while ((m = re.exec(txt))) used.add(m[1]);
  }
}

const ex = parseDotEnv(path.join(ROOT, ".env.example"));
const missing = [...used].filter((k) => !ex.has(k)).sort();

console.log("== ENV AUDIT ==");
console.log(`Detected env keys used in code: ${used.size}`);
if (!used.size) {
  console.log("NOTE: No env keys detected by regex; auth/DB may still require env.");
  process.exit(0);
}

if (missing.length) {
  console.log("FAIL: Missing in .env.example:");
  missing.forEach((k) => console.log(`  - ${k}`));
  process.exit(1);
}
console.log("PASS: .env.example covers detected env keys.");

/**
 * Route map (best-effort) for Next.js (pages/app).
 */
const fs = require("fs");
const path = require("path");

const ROOT = path.resolve(__dirname, "..");
const EXCLUDE = new Set(["node_modules", ".git", "dist", "build", ".next", ".cache"]);

function walk(dir, out = []) {
  if (!fs.existsSync(dir)) return out;
  for (const ent of fs.readdirSync(dir, { withFileTypes: true })) {
    if (ent.isDirectory()) {
      if (EXCLUDE.has(ent.name)) continue;
      walk(path.join(dir, ent.name), out);
    } else if (ent.isFile()) {
      if (/\.(js|jsx|ts|tsx|mdx)$/.test(ent.name)) out.push(path.join(dir, ent.name));
    }
  }
  return out;
}

function inferRoutes(base) {
  const routes = new Set();
  for (const f of walk(base)) {
    const rel = path.relative(base, f).replace(/\\/g, "/");
    if (rel.startsWith("_")) continue;
    if (rel.includes("[") && rel.includes("]")) continue;
    let r = rel.replace(/\.(js|jsx|ts|tsx|mdx)$/, "");
    r = r.replace(/\/(page|index)$/, "");
    r = r === "index" ? "" : r;
    routes.add("/" + r);
  }
  return routes;
}

const bases = [
  path.join(ROOT, "pages"),
  path.join(ROOT, "src", "pages"),
  path.join(ROOT, "app"),
  path.join(ROOT, "src", "app"),
].filter(fs.existsSync);

console.log("== ROUTE MAP ==");
if (!bases.length) {
  console.log("NOTE: No Next.js pages/app directories detected.");
  process.exit(0);
}

const all = new Set(["/"]);
for (const b of bases) for (const r of inferRoutes(b)) all.add(r.replace(/\/+$/, "") || "/");

[...all].sort().forEach((r) => console.log(r));


--- tools/link_scan.js ---
/**
 * Link scan: finds internal href="/..." and flags ones not in route map (best-effort).
 */
const fs = require("fs");
const path = require("path");

const ROOT = path.resolve(__dirname, "..");
const EXCLUDE = new Set(["node_modules", ".git", "dist", "build", ".next", ".cache"]);
const LINK_RE = /href\s*=\s*["'](\/[^"']*)["']/g;

function walk(dir, out = []) {
  for (const ent of fs.readdirSync(dir, { withFileTypes: true })) {
    if (ent.isDirectory()) {
      if (EXCLUDE.has(ent.name)) continue;
      walk(path.join(dir, ent.name), out);
    } else if (ent.isFile()) {
      const p = path.join(dir, ent.name);
      if (/\.(js|jsx|ts|tsx|md|mdx|html)$/.test(p)) out.push(p);
    }
  }
  return out;
}

function routeSet() {
  const candidates = [
    path.join(ROOT, "pages"),
    path.join(ROOT, "src", "pages"),
    path.join(ROOT, "app"),
    path.join(ROOT, "src", "app"),
  ].filter(fs.existsSync);

  const routes = new Set(["/"]);
  for (const base of candidates) {
    const files = [];
    const w = (d) => {
      for (const ent of fs.readdirSync(d, { withFileTypes: true })) {
        const p = path.join(d, ent.name);
        if (ent.isDirectory()) w(p);
        else if (ent.isFile() && /\.(js|jsx|ts|tsx|mdx)$/.test(ent.name)) files.push(p);
      }
    };
    w(base);
    for (const f of files) {
      const rel = path.relative(base, f).replace(/\\/g, "/");
      if (rel.startsWith("_")) continue;
      if (rel.includes("[") && rel.includes("]")) continue;
      let r = rel.replace(/\.(js|jsx|ts|tsx|mdx)$/, "");
      r = r.replace(/\/(page|index)$/, "");
      r = r === "index" ? "" : r;
      routes.add("/" + r);
    }
  }
  return routes;
}

const routes = routeSet();
const seen = new Map();

for (const f of walk(ROOT)) {
  const txt = fs.readFileSync(f, "utf8");
  let m;
  while ((m = LINK_RE.exec(txt))) {
    const href = m[1].split("#")[0].split("?")[0];
    if (href.startsWith("/api")) continue;
    if (!seen.has(href)) seen.set(href, new Set());
    seen.get(href).add(path.relative(ROOT, f));
  }
}

const suspicious = [];
for (const [href, files] of seen.entries()) {
  if (!routes.has(href) && href !== "/") suspicious.push({ href, files: [...files] });
}

console.log("== LINK SCAN ==");
console.log(`hrefs: ${seen.size} | routes: ${routes.size}`);
if (!suspicious.length) {
  console.log("PASS: No obvious dead internal links (best-effort).");
  process.exit(0);
}
console.log("FAIL: Potential dead links:");
suspicious.slice(0, 80).forEach(({ href, files }) => {
  console.log(`- ${href}`);
  files.slice(0, 5).forEach((f) => console.log(`   • ${f}`));
  if (files.length > 5) console.log(`   • (+${files.length - 5} more)`);
});
process.exit(1);


--- tools/auth_smoke.js ---
/**
 * Auth smoke: best-effort checks without knowing your exact stack.
 * - Looks for common auth routes and prints guidance.
 * - If NEXT_PUBLIC_APP_URL exists, attempts fetch.
 */
const http = require("http");
const https = require("https");

const base = process.env.NEXT_PUBLIC_APP_URL || process.env.APP_URL || "http://localhost:3000";
const candidates = [
  "/api/auth/session",
  "/api/auth/signin",
  "/api/auth/callback",
  "/api/login",
  "/api/admin",
];

function get(url) {
  return new Promise((resolve) => {
    const lib = url.startsWith("https") ? https : http;
    const req = lib.get(url, (res) => {
      res.resume();
      resolve({ url, status: res.statusCode || 0 });
    });
    req.on("error", () => resolve({ url, status: 0 }));
    req.setTimeout(2500, () => {
      req.destroy();
      resolve({ url, status: 0 });
    });
  });
}

(async () => {
  console.log("== AUTH SMOKE ==");
  console.log(`Base: ${base}`);
  const results = [];
  for (const p of candidates) results.push(await get(base + p));

  for (const r of results) {
    const ok = [200, 302, 401, 403].includes(r.status);
    console.log(`${ok ? "OK " : "?? "} ${r.status} ${r.url}`);
  }

  console.log("");
  console.log("Interpretation:");
  console.log("- 200/302: route exists and responds.");
  console.log("- 401/403: protected/unauthenticated behavior is working.");
  console.log("- 0: server not reachable OR route missing.");
  process.exit(0);
})();
